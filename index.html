    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, collection, doc, setDoc, onSnapshot, deleteDoc, getDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // CONFIGURATION FIREBASE (À REMPLACER PAR TES CLÉS DANS GITHUB)

        const configStr = typeof __firebase_config !== 'undefined' ? __firebase_config : '{"apiKey":""}';

        const firebaseConfig = JSON.parse(configStr);

        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'f16-barber-v1';


        let currentUser = null;

        let appointments = [], blockedDays = [], scheduleConfig = { start: 10, end: 20 }, selectedDate = null, selectedSlot = null;

        let currentMonth = new Date().getMonth(), currentYear = new Date().getFullYear();


        const initApp = async () => {

            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                try { await signInWithCustomToken(auth, __initial_auth_token); } 

                catch (e) { await signInAnonymously(auth); }

            } else {

                await signInAnonymously(auth);

            }

        };


        onAuthStateChanged(auth, (u) => {

            currentUser = u;

            if (u) initRealtime();

        });


        initApp();


        const apptsRef = collection(db, 'artifacts', appId, 'public', 'data', 'appointments');

        const blockedRef = collection(db, 'artifacts', appId, 'public', 'data', 'blockedDays');

        const configDocRef = doc(db, 'artifacts', appId, 'public', 'data', 'config', 'schedule');


        function initRealtime() {

            if (!currentUser) return;

            onSnapshot(apptsRef, (snap) => {

                appointments = snap.docs.map(d => ({ id: d.id, ...d.data() }));

                renderSlots();

                renderAdminView();

            }, (e) => console.error(e));


            onSnapshot(blockedRef, (snap) => {

                blockedDays = snap.docs.map(d => d.data().date);

                renderCalendar();

                renderSlots();

                renderAdminBlocked();

            }, (e) => console.error(e));


            onSnapshot(configDocRef, (snap) => {

                if (snap.exists()) {

                    scheduleConfig = snap.data();

                    renderSlots();

                }

            }, (e) => console.error(e));

        }


        async function hash(s) { const b = new TextEncoder().encode(s); const h = await crypto.subtle.digest('SHA-256', b); return Array.from(new Uint8Array(h)).map(x => x.toString(16).padStart(2, '0')).join(''); }

        document.getElementById('adminLoginBtn').onclick = async () => {

            const p = document.getElementById('adminPassInput').value; if (!p) return;

            const h = await hash(p); const st = localStorage.getItem('f16_h');

            if (!st) { localStorage.setItem('f16_h', h); unlock(); } else if (h === st) unlock(); else showToast("FAUX");

        };

        function unlock() { sessionStorage.setItem('f16_in', '1'); document.getElementById('adminAuthPanel').classList.add('hidden'); document.getElementById('adminControls').classList.remove('hidden'); document.getElementById('adminLogoutBtn').classList.remove('hidden'); renderAdminView(); renderAdminBlocked(); }

        document.getElementById('adminLogoutBtn').onclick = () => { sessionStorage.removeItem('f16_in'); location.reload(); };

        

        document.getElementById('saveSchedule').onclick = async () => {

            if (!currentUser) return;

            await setDoc(configDocRef, { start: +document.getElementById('startHourInput').value, end: +document.getElementById('endHourInput').value });

            showToast("HORAIRES MAJ");

        };

        

        document.getElementById('addBlockedDay').onclick = async () => {

            if (!currentUser) return;

            const d = document.getElementById('blockDatePicker').value;

            if (d) await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'blockedDays', d), { date: d });

        };

        

        window.unblockDay = async (d) => {

            if (currentUser) await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'blockedDays', d));

        };

        

        window.cancelAppt = async (id) => {

            if (currentUser) await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'appointments', id));

        };


        document.getElementById('confirmBooking').onclick = async () => {

            if (!currentUser) return;

            const n = document.getElementById('clientName').value; if (!n || !selectedSlot) return showToast("INFOS ?");

            await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'appointments', crypto.randomUUID()), { date: selectedDate, time: selectedSlot, clientName: n, createdAt: new Date().toISOString() });

            showToast("RDV PRIS !"); document.getElementById('clientName').value = ""; document.getElementById('selectionSummary').classList.add('opacity-0'); document.getElementById('emptySummary').classList.remove('hidden');

        };


        function renderCalendar() {

            const grid = document.getElementById('calendarGrid');

            const monthNames = ["Janvier", "Février", "Mars", "Avril", "Mai", "Juin", "Juillet", "Août", "Septembre", "Octobre", "Novembre", "Décembre"];

            document.getElementById('currentMonthYear').textContent = `${monthNames[currentMonth]} ${currentYear}`;

            grid.innerHTML = '';

            const first = new Date(currentYear, currentMonth, 1).getDay();

            const days = new Date(currentYear, currentMonth + 1, 0).getDate();

            let off = first === 0 ? 6 : first - 1;

            for (let i = 0; i < off; i++) grid.innerHTML += '<div class="calendar-day disabled"></div>';

            const today = new Date(); today.setHours(0,0,0,0);

            for (let d = 1; d <= days; d++) {

                const ds = `${currentYear}-${String(currentMonth + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;

                const bl = blockedDays.includes(ds), ps = new Date(currentYear, currentMonth, d) < today;

                grid.innerHTML += `<div class="calendar-day ${bl || ps ? 'disabled' : ''} ${selectedDate === ds ? 'selected' : ''}" onclick="window.selectDate('${ds}')">${d}</div>`;

            }

        }


        window.selectDate = (ds) => { 

            selectedDate = ds; 

            document.getElementById('calendarContainer').classList.add('scanning'); 

            setTimeout(() => { 

                document.getElementById('calendarContainer').classList.remove('scanning'); 

                renderCalendar(); 

                renderSlots(); 

                document.getElementById('slotsWrapper').classList.remove('opacity-0', 'translate-y-4'); 

            }, 500); 

        };


        function renderSlots() {

            const cont = document.getElementById('slotsContainer'), st = document.getElementById('statusMessage'); if (!selectedDate) return;

            if (blockedDays.includes(selectedDate)) { cont.innerHTML = ""; st.textContent = "FERMÉ CE JOUR."; st.className = "mt-6 p-4 bg-red-600 text-white font-bold text-center block rounded"; document.getElementById('slotsWrapper').classList.add('opacity-0'); return; }

            st.className = "hidden"; const sls = []; for (let h = scheduleConfig.start; h < scheduleConfig.end; h++) { sls.push(`${h}:00`, `${h}:30`); }

            cont.innerHTML = sls.map(t => {

                const bk = appointments.some(a => a.date === selectedDate && a.time === t);

                return `<button class="slot-btn border border-white/10 rounded py-4 text-[11px] font-bold ${bk ? 'text-white/5 bg-neutral-800' : ''}" ${bk ? 'disabled' : `onclick="window.handleSlotSelection('${t}')"`} data-time="${t}">${t}</button>`;

            }).join('');

        }


        window.handleSlotSelection = (t) => {

            selectedSlot = t; document.querySelectorAll('.slot-btn').forEach(b => b.classList.remove('bg-red-600', 'text-white'));

            const b = document.querySelector(`.slot-btn[data-time="${t}"]`); if (b) b.classList.add('bg-red-600', 'text-white');

            document.getElementById('emptySummary').classList.add('hidden'); document.getElementById('selectionSummary').classList.remove('opacity-0', 'translate-y-4');

            document.getElementById('summaryTime').textContent = t; const [y,m,d] = selectedDate.split('-');

            document.getElementById('summaryDate').textContent = new Date(y, m-1, d).toLocaleDateString('fr-FR', { weekday: 'long', day: 'numeric', month: 'long' }).toUpperCase();

        };


        function renderAdminBlocked() { if (sessionStorage.getItem('f16_in') !== '1') return; document.getElementById('blockedDaysList').innerHTML = blockedDays.map(d => `<div class="flex justify-between bg-black p-2 mb-2"><span class="text-[10px] uppercase font-bold text-white">${d}</span><button onclick="window.unblockDay('${d}')" class="text-red-500 font-bold">X</button></div>`).join(''); }

        function renderAdminView() { if (sessionStorage.getItem('f16_in') !== '1') return; const s = [...appointments].sort((a,b) => a.date.localeCompare(b.date) || a.time.localeCompare(b.time)); document.getElementById('adminAppointmentsList').innerHTML = s.map(a => `<div class="bg-black/80 p-4 rounded border border-white/10 flex justify-between items-center text-white"><div><p class="font-bold text-xs uppercase">${a.clientName}</p><p class="text-red-500 text-[10px] font-bold uppercase">${a.date} — ${a.time}</p></div><button onclick="window.cancelAppt('${a.id}')" class="text-red-500 text-[10px] font-bold">VIRER</button></div>`).join('') || '<p class="text-white/20 text-[10px] italic">Aucun rdv.</p>'; }


        document.getElementById('prevMonth').onclick = () => { currentMonth--; if (currentMonth < 0) { currentMonth = 11; currentYear--; } renderCalendar(); };

        document.getElementById('nextMonth').onclick = () => { currentMonth++; if (currentMonth > 11) { currentMonth = 0; currentYear++; } renderCalendar(); };

        if (sessionStorage.getItem('f16_in') === '1') unlock(); renderCalendar();


        function reveal() { document.querySelectorAll(".reveal").forEach(r => { if (r.getBoundingClientRect().top < window.innerHeight - 100) r.classList.add("active"); }); }

        window.addEventListener("scroll", reveal); window.onload = reveal;

        window.scrollToSection = (id) => document.getElementById(id).scrollIntoView({ behavior: 'smooth' });

        function showToast(m) { const t = document.getElementById('toast'); t.textContent = m; t.style.opacity = '1'; t.style.transform = 'translate(-50%, 0)'; setTimeout(() => { t.style.opacity = '0'; t.style.transform = 'translate(-50%, 40px)'; }, 3000); }

        window.toggleAdmin = (s) => document.getElementById('adminModal').classList.toggle('hidden', !s);

        document.getElementById('adminToggle').onclick = () => toggleAdmin(true);

    </script>
